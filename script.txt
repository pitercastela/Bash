menu () {
while true
    do
    cabecalho
echo "Menu de escolhas: 

    1)
    2)
    3)
    4)
    5) Finalizar programa."
    
    
    read escolha
    case $escolha in

        1)
            adicionar_remover_arquivos
            ;;

        2)
            cat biblioteca_log.txt
            ;;

        3)
            echo "3
            "
            ;;

        4)
            echo "4
            "
            ;;
    
        5)
            trap 2
            break
            ;;    

        *)
            echo "Opção inválida"
            sleep 1
            ;;

esac
    done
}

adicionar_remover_arquivos () {
    echo
    cabecalho
    
    while true
    do
echo "Menu de escolhas: 

    1) Adicionar
    2) Remover
    3) Finalizar programa."
    
    
    read escolha
    case $escolha in

        1)
            sleep 1
            adicionar
            ;;

        2)
            echo "2
            "
            ;;

        3)
            echo "3
            "
            break
            ;;

        *)
            echo "Opção inválida"
            sleep 1
            ;;

esac
    done
    
    
}

adicionar() {
    echo "Digite o nome do livro."
    read livro
    echo "Digite o nome do autor do livro."
    read autor
    echo "Digite uma descrição breve do livro"
    read desc
    echo "Digite a quantidade em estoque do livro"
    read qtd
    
    echo "[LIVRO]: $livro | [AUTOR]: $autor | [DESC]: $desc | [QTD]: $qtd" >> biblioteca_log.txt
}






menu
